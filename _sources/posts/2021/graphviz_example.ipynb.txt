    "\n",
    "# Start counting at one for node numbers\n",
    "num_node = 1\n",
    "\n",
    "# Loop through the different experiments\n",
    "for experiment in df.experiment.unique():\n",
    "    exp_i = num_node\n",
    "    dot.node(str(exp_i), label=experiment)\n",
    "    num_node += 1\n",
    "\n",
    "    # Loop through the different components in each experiment\n",
    "    for component in df.loc[df.experiment == experiment].component.unique():\n",

    "        dot.node(str(comp_i), label=component)\n",
